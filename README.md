# Neo Vim Config
  
    First install repository: https://github.com/junegunn/vim-plug
    Copy init.vim to your OS: ~/.config/nvim/init.vim
    
    run nvim
    :PlugInstall
    :PlugClean
    
    sudo apt install nodejs
    sudo apt-get install npm
    
    cd ~/.local/share/nvim/plugged/coc.nvim/
    npm install
